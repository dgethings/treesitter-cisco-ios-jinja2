===
config-router rich negation
===

router ospf 1
 no neighbor 10.0.0.2
 no network 10.0.0.0 0.0.0.255 area 0
 no passive-interface GigabitEthernet0/0
 no redistribute bgp
!

---

(config
  (router_section
    (router_header
      protocol: (routing_protocol)
      process_id: (value))
    (negated_statement
      (negation_keyword)
      keyword: (neighbor_statement
        arg: (value)))
    (negated_statement
      (negation_keyword)
      keyword: (network_statement
        arg: (value)
        arg: (value)
        arg: (value)
        arg: (value)))
    (negated_statement
      (negation_keyword)
      keyword: (passive_interface_statement
        arg: (value)))
    (negated_statement
      (negation_keyword)
      keyword: (redistribute_statement
        arg: (value)))
    (eos)))
